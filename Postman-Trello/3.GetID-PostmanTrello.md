# 📌 GET ID for the "Postman-Trello" Board

This request fetches all Trello boards for the authenticated user and extracts the ID of the one named **"Postman-Trello"**.

---

## ✅ Request URL

```http
GET https://api.trello.com/1/members/me/boards?key={{apiKey}}&token={{apiToken}}
```

---

##  Test Script

```javascript
// Extract board list from JSON response
const boards = pm.response.json();

// Search for board named "Postman-Trello"
const board = boards.find(b => b.name === "Postman-Trello");

if (board) {
    // Set ID in environment for reuse
    pm.environment.set("board_id", board.id);

    // ✅ Display board name in test result
    pm.test("Board Name: " + board.name, function () {
        pm.expect(board.name).to.eql("Postman-Trello");
    });

    // ✅ Display board ID in test result
    pm.test("Board ID: " + board.id, function () {
        pm.expect(board.id).to.be.a("string");
    });

    // Optional: Verify board exists
    pm.test("Board exists", function () {
        pm.expect(board).to.not.be.undefined;
    });
} else {
    // If not found, show test failure
    pm.test("Board 'Postman-Trello' not found", function () {
        pm.expect.fail("Board named 'Postman-Trello' was not found.");
    });
}
```
